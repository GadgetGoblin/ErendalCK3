tradition_dwarven_military = {

	category = regional

		is_shown = {
		has_cultural_pillar = heritage_dwarven
		}
	
		can_pick = {
		has_cultural_pillar = heritage_dwarven
		}
	}
		parameters = {
		unlock_maa_younglings = yes
		unlock_maa_dwarven_crossbowmen = yes
		unlock_maa_boar_riders = yes
		unlock_maa_oathsworn = yes
		unlock_maa_longbeards = yes
		unlock_maa_dwarven_muskets = yes
		unlock_maa_dwarven_house_guard = yes
		
	}
	
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bellicose = { is_in_list = traits }
						culture_pillar:ethos_courtly = { is_in_list = traits }
						culture_pillar:ethos_communal = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_courtly_or_communal_desc
				}
			}
			if = {
				limit = {
					NOT = {
						scope:character = {
							any_knight = {
								count >= 6
								prowess >= 12
							}
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = knights_with_prowess_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}

	ai_will_do = { value = 0 }
}
